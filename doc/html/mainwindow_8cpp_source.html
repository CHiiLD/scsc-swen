222 bool extractDataset(QAbstractItemModel* model, int col, dataset& data)
223 {
224  int row, rowCount = model->rowCount();
225  bool ok = false;
226  dataset list;
227  // 공백을 만날 때까지 테이블뷰에서 데이터를 수집한다.
228  for(row = 0; row < rowCount; ++row)
229  {
230  QModelIndex index = model->index(row, col);
231  QVariant var = model->data(index);
232  if(!var.isValid() || var.toString().length() == 0) // 공백을 만난 경우
233  break;
234 
235  double value = var.toDouble(&ok);
236  if(!ok) // 잘못된 데이터 입력을 감지
237  break;
238 
239  list.push_back(value);
240  }
241  // 공백 이후에 데이터가 추가로 입력된 경우를 찾아서 잘못된 데이터 입력으로 처리
242  for(++row; row < rowCount; ++row)
243  {
244  QModelIndex index = model->index(row, col);
245  QVariant var = model->data(index);
246  if(var.isValid() && var.toString().length() != 0)
247  {
248  ok = false;
249  break;
250  }
251  }
252  if(ok)
253  data = list;
254  return ok;
255 }

474 void MainWindow::judgeNullHypothesis(bool checked)
475 {
476  QString sTestStatistic = ui->lineEdit_nullHypothesisTestStatistic->text();
477  QString sStdDevation = ui->lineEdit_nullHypothesisStdDeviation->text();
478  QString sSize = ui->lineEdit_nullHypothesisSize->text();
479  QString sTestValue = ui->lineEdit_nullHypothesisTestValue->text();
480  QString sAlpha = ui->lineEdit_nullHypothesisAlpha->text();
481  QString sSign = ui->comboBox_sign->currentText();
482  double testStatistic, stdDevation, testValue, alpha;
483  int size;
484  NullHypothesisSign sign = NullHypothesisSign::NONE;
485  bool ok = false;
486  do
487  {
488  testStatistic = sTestStatistic.toDouble(&ok);
489  if(!ok) break;
490  stdDevation = sStdDevation.toDouble(&ok);
491  if(!ok) break;
492  testValue = sTestValue.toDouble(&ok);
493  if(!ok) break;
494  alpha = sAlpha.toDouble(&ok);
495  if(!ok) break;
496  size = sSize.toInt(&ok);
497  if(!ok) break;
498  } while(false);
499 
500  if(sSign.compare(NULL_HYPOTHESIS_SIGN_NEIN) == 0)
501  sign = NullHypothesisSign::NOT_EQUAL;
502  else if (sSign.compare(NULL_HYPOTHESIS_SIGN_LESS_THEN))
503  sign = NullHypothesisSign::LESS_THEN;
504  else if (sSign.compare(NULL_HYPOTHESIS_SIGN_GREATER_THEN))
505  sign = NullHypothesisSign::GREATER_THEN;
506 
507  if(!ok || sign == NullHypothesisSign::NONE)
508  {
509  showWrongInputMessageBox();
510  return;
511  }
512 
513  int ret = NullHypothesis::hyphothesis(testStatistic, stdDevation, size, testValue, alpha, sign);
514  QString text;
515  text.append("α=");
516  text.append(QString::number(alpha));
517  text.append("에서 귀무가설을 ");
518  switch (ret)
519  {
520  case NULL_HYPOTHESIS_REJECT_SECCESS:
521  text.append("기각한다.");
522  break;
523  case NULL_HYPOTHESIS_REJECT_FAIL:
524  text.append("기각 할 수 없다.");
525  break;
526  default:
527  showWrongInputMessageBox();
528  return;
529  }
530  ui->textEdit_nullHypothesisOutput->setText(text);
531 }
